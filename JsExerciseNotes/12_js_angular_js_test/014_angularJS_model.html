AngularJS 模块

AngularJS 模块是一个包含了应用程序中所有指令、过滤器、服务、模块的命名空间

模块创建

angular.module('myApp', []);

模块配置

angular.module('myApp', []).config(['$httpProvider', function($httpProvider ) { $httpProvider.interceptors.push('myInterceptor'); }]);

模块指令

angular.module('myApp', []).directive('myDirective', function() { return { restrict: 'A', template: '

Directive works!
' }; });

模块过滤器

angular.module('myApp', []).filter('myFilter', function() { return function(input) { return 'filtered:'+ input; }; });

模块控制器

angular.module('myApp', []).controller('myCtrl', function($scope) { $scope.message = 'Hello World'; });
